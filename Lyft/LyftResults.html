            strides=(4,4), padding='same',
            name="Output_Channels")(X)
B_C = Conv2DTranspose(24, (3,3), 
            activation='tanh', 
            strides=(4,4), padding='same',
            name="Box_Corners")(X)
model = Model(
            inputs=inputs,
            outputs=[O_C, B_C],
            name="Lyft")
